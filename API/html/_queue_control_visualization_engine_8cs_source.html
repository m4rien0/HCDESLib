133  public void DrawHoldingEntity(IDynamicHoldingEntity holdingEntity)
134  {
135  DrawDynamicHoldingEntity drawFoHold = (DrawDynamicHoldingEntity)DrawingObjectPerEntity((Entity)holdingEntity);
136 
137  if (drawFoHold.MaxNumberPerson < holdingEntity.HoldedEntities.Count)
138  {
139  drawFoHold.DrawPersonCount.StringToDraw = holdingEntity.HoldedEntities.Count.ToString();
140 
141  if (!DrawingSystem.DrawingObjects.Contains(drawFoHold.DrawPersonCount))
142  DrawingSystem.AddObject(drawFoHold.DrawPersonCount);
143 
144  foreach (DrawingObject drawObject in holdingEntity.HoldedEntities.Select(p => DrawingObjectPerEntity(p)))
145  {
146  drawObject.SetPosition(drawFoHold.DrawPersonCount.CurrentPosition - new Vector(-30, 0));
147  } // end foreach
148  }
149  else
150  {
151  if (DrawingSystem.DrawingObjects.Contains(drawFoHold.DrawPersonCount))
152  DrawingSystem.RemoveObject(drawFoHold.DrawPersonCount);
153 
154  int entitiesDrawn = 0;
155  double entityW = drawFoHold.SlotWidth;
156  double entityH = drawFoHold.SlotHeight;
157 
158  for (int i = 0; i < drawFoHold.NumberPersonVertical; i++)
159  {
160  for (int j = 0; j < drawFoHold.NumberPersonHorizontal; j++)
161  {
162  if (entitiesDrawn == holdingEntity.HoldedEntities.Count)
163  return;
164 
165  DrawingObject drawEntity = DrawingObjectPerEntity(holdingEntity.HoldedEntities[entitiesDrawn]);
166 
167  drawEntity.SetPosition(drawFoHold.CurrentPosition + new Vector(j * entityW + drawFoHold.SlotWidth / 2, i * entityH));
168 
169  entitiesDrawn++;
170  } // end for
171  } // end for
172  } // end if
173  } // end of DrawHoldingEntity
