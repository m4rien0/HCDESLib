252  public void DrawHoldingEntity(IDynamicHoldingEntity holdingEntity)
253  {
254  DrawDynamicHoldingEntity drawFoHold = (DrawDynamicHoldingEntity)DrawingObjectPerEntity((Entity)holdingEntity);
255 
256  if (drawFoHold.MaxNumberPerson < holdingEntity.HoldedEntities.Count)
257  {
258  drawFoHold.DrawPersonCount.StringToDraw = holdingEntity.HoldedEntities.Count.ToString();
259 
260  if (!DrawingSystem.DrawingObjects.Contains(drawFoHold.DrawPersonCount))
261  DrawingSystem.AddObject(drawFoHold.DrawPersonCount);
262 
263  foreach (DrawingObject drawObject in holdingEntity.HoldedEntities.Select(p=> DrawingObjectPerEntity(p)))
264  {
265  drawObject.SetPosition(drawFoHold.DrawPersonCount.CurrentPosition - new Vector(-PersonSize, 0));
266  } // end foreach
267  }
268  else
269  {
270  if (DrawingSystem.DrawingObjects.Contains(drawFoHold.DrawPersonCount))
271  DrawingSystem.RemoveObject(drawFoHold.DrawPersonCount);
272 
273  int entitiesDrawn = 0;
274  double entityW = drawFoHold.SlotWidth;
275  double entityH = drawFoHold.SlotHeight;
276 
277  for (int i = 0; i < drawFoHold.NumberPersonVertical; i++)
278  {
279  for (int j = 0; j < drawFoHold.NumberPersonHorizontal; j++)
280  {
281  if (entitiesDrawn == holdingEntity.HoldedEntities.Count)
282  return;
283 
284  DrawingObject drawEntity = DrawingObjectPerEntity(holdingEntity.HoldedEntities[entitiesDrawn]);
285 
286  drawEntity.SetPosition(drawFoHold.CurrentPosition + new Vector(j * entityW + drawFoHold.SlotWidth / 2, i * entityH));
287 
288  entitiesDrawn++;
289  } // end for
290  } // end for
291  } // end if
292 
293  } // end of DrawHoldingEntity

392  drawMainNurse.SetPosition(drawTreatFac.GetStaffPosition(staffCount));
393  staffCount++;
394  } // end if

399  {
400  DrawDoctor drawAssistDoctor = (DrawDoctor)DrawingObjectPerEntity(ent);
