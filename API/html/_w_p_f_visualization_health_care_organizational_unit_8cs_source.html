153  public void DrawHoldingEntity(IDynamicHoldingEntity holdingEntity)
154  {
155  DrawDynamicHoldingEntity drawFoHold = (DrawDynamicHoldingEntity)DrawingObjectPerEntity((Entity)holdingEntity);
156 
157  if (drawFoHold.MaxNumberPerson < holdingEntity.HoldedEntities.Count)
158  {
159  drawFoHold.DrawPersonCount.StringToDraw = holdingEntity.HoldedEntities.Count.ToString();
160 
161  if (!DrawingSystem.DrawingObjects.Contains(drawFoHold.DrawPersonCount))
162  DrawingSystem.AddObject(drawFoHold.DrawPersonCount);
163 
164  foreach (DrawingObject drawObject in holdingEntity.HoldedEntities.Select(p=> DrawingObjectPerEntity(p)))
165  {
166  drawObject.SetPosition(drawFoHold.DrawPersonCount.CurrentPosition - new Vector(-PersonSize, 0));
167  } // end foreach
168  }
169  else
170  {
171  if (DrawingSystem.DrawingObjects.Contains(drawFoHold.DrawPersonCount))
172  DrawingSystem.RemoveObject(drawFoHold.DrawPersonCount);
173 
174  int entitiesDrawn = 0;
175  double entityW = drawFoHold.SlotWidth;
176  double entityH = drawFoHold.SlotHeight;
177 
178  for (int i = 0; i < drawFoHold.NumberPersonVertical; i++)
179  {
180  for (int j = 0; j < drawFoHold.NumberPersonHorizontal; j++)
181  {
182  if (entitiesDrawn == holdingEntity.HoldedEntities.Count)
183  return;
184 
185  DrawingObject drawEntity = DrawingObjectPerEntity(holdingEntity.HoldedEntities[entitiesDrawn]);
186 
187  drawEntity.SetPosition(drawFoHold.CurrentPosition + new Vector(j * entityW + drawFoHold.SlotWidth / 2, i * entityH));
188 
189  entitiesDrawn++;
190  } // end for
191  } // end for
192  } // end if
193 
194  } // end of DrawHoldingEntity
